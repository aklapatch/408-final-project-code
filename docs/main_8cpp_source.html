59 string textBetween(string input, const char *startchars, const char *endchars) {
60 
61  // check if the string is big engough to actually filter
62  // this will throw and exception if we do not do this
63  size_t filter_size = strlen(startchars) + strlen(endchars);
64  if (filter_size >= input.size())
65  return "";
66 
67  size_t start_idx = input.find(startchars);
68  size_t end_idx = input.find(endchars);
69 
70  // return an empty string if you cannot find the filter text
71  if (start_idx != string::npos || end_idx != string::npos)
72  return "";
73 
74  if (start_idx > end_idx)
75  return "";
76 
77  start_idx += strlen(startchars);
78 
79  return input.substr(start_idx, end_idx - start_idx);
80 }
